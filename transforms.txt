class State:

	def __init__(self):
		self.stack = []
		self.space = False
		self.space_before = None

	def handle_string(self, node):
		match self.space:
			case True:
				node = squeeze(node.data.rstrip())
				if len(node) == 0:
					return
				self.stack.append(" ")
				self.stack.append(node)
				self.space = None
			case _:
				if self.space_before is None and len(node) > 0:
					self.space_before = node[0].isspace()
				node = squeeze(node.data.lstrip())
				if len(node) == 0:
					return
				if node[-1].isspace():
					node = node[:-1]
					self.stack.append(node)
					self.space = True
				self.stack.append(node)
				self.space = None

	def handle_tag(self, node):
		match self.space:
			case True:
				left_space, repl, right_space = cleanup_tag(node)
				if left_space is not False:
					stack.append(" ")
				if repl:
					stack.append(repl)
				return right_space
			case False:
				_, repl, right_space = cleanup_tag(node)
				if repl:
					stack.append(repl)
				return right_space
			case None:
				left_space, repl, right_space = cleanup_tag(node)
				if left_space:
					stack.append(" ")
				if repl:
					stack.append(repl)
				return right_space
